                    const setClock = () => {
                        let date = new Date();
                        let datearr = [ date.getFullYear(), (date.getMonth() + 1), date.getDate(), date.getDay(), date.getHours(), date.getMinutes(), date.getSeconds() ];

                        let arrid = ["year", "month", "day", "weekend", "hours", "minute", "second"]
                        let days = ["일", "월", "화", "수", "목", "금", "토"]
                        let houridx = 4
                        let weekendidx = 3


                        let target = document.getElementById("t_area")
                        target.innerText = (12 > datearr[houridx]) ? '오전' : '오후'
                        for( let i = 0; i < arrid.length; i++) {
                            let values = datearr[i]
                            target = document.getElementById(arrid[i])
                            target.innerText = ( houridx === i ) ? ( ( 12 > values ) ? values : values - 12 ) : ( ( weekendidx === i ) ? days[values] : values );
                        }
                    }

                    const Call_Season = () => {
                        let idate = new Date();
                        let month = (idate.getMonth() + 1); // 1 : monthidx
                        let alert_val = ( 3 <= month && 5 >= month ) ? '봄의 계절' : ( 6 <= month && 8 >= month ) ? '여름의 계절' : ( 9 <= month && 11 >= month ) ? '가을의 계절' : ( 12 <= month && 2 >= month ) ? '겨울의 계절' : -1;
                        if ( 0 < month ) {
                            alert( alert_val );
                        }
                    }

                    document.getElementById("season_check").addEventListener("click", Call_Season);

                    

                    window.onload = () => {
                        setClock();
                        setInterval(setClock, 1000);
                    }
